! ==========================
! SW-IT - VLAN 10 Access SW
! ==========================

enable
configure terminal

hostname SW-IT
no ip domain-lookup

vlan 10
 name IT
exit

! PCs on access ports (edit ranges if needed)
interface range fastEthernet0/1-3
 switchport mode access
 switchport access vlan 10
 spanning-tree portfast
exit

! Uplink to MLS (trunk)
interface fastEthernet0/4
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30
 no shutdown
exit

end
write memory

! --- Verification ---
! show vlan brief
! show interfaces trunk
